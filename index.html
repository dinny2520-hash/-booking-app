ขั้นตอนการตั้งค่า Apps Script

  1. เปิด Google Sheets ใหม่ → Extensions → Apps Script
  2. ลบโค้ดเดิมออก แล้ววางโค้ดด้านล่างทั้งหมด
  3. Save → Deploy → New deployment
  4. Type: Web app · Execute as: Me · Who has access: Anyone
  5. กด Deploy → Authorize → คัดลอก Web app URL
  6. วาง URL ในหน้าผู้ดูแล แล้วกดบันทึก
const SHEET_NAME = "Bookings"; const SLOTS_SHEET = "Slots"; function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const type = e.parameter.type; const action = e.parameter.action; const cb = e.parameter.callback; // JSONP support if (type === "slots") { const sh = ss.getSheetByName(SLOTS_SHEET); if (!sh || sh.getLastRow() < 2) return out([], cb); const rows = sh.getRange(2,1,sh.getLastRow()-1,8).getValues(); return out(rows.map(r=>({ id:r[0], name:r[1], date:fmtDate(r[2]), start:fmtTime(r[3]), end:fmtTime(r[4]), type:r[5], cap:Number(r[6]), location:r[7] })), cb); } if (type === "bookings") { const sh = ss.getSheetByName(SHEET_NAME); if (!sh || sh.getLastRow() < 2) return out([], cb); const rows = sh.getRange(2,1,sh.getLastRow()-1,9).getValues(); return out(rows.map(r=>({ id:r[0], empId:r[1], firstName:r[2], lastName:r[3], branch:r[4], slotId:r[5], slotName:r[6], date:fmtDate(r[7]), time:String(r[8]) })), cb); } if (action && e.parameter.data) { const data = JSON.parse(decodeURIComponent(e.parameter.data)); handleAction(ss, action, data); return out({success:true}, cb); } return out({error:"unknown"}, cb); } function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const data = JSON.parse(e.postData.contents); handleAction(ss, data.action, data); return out({success:true}, null); } function handleAction(ss, action, data) { if (action === "book") { const sh = getOrCreate(ss, SHEET_NAME, ["ID","รหัสพนักงาน","ชื่อจริง","นามสกุล","สาขา","SlotID","ชื่อรอบ","วันที่","เวลา"]); sh.appendRow([data.id,data.empId||'',data.firstName,data.lastName,data.branch, data.slotId,data.slotName,data.date,data.time]); } if (action === "cancel") { const sh = ss.getSheetByName(SHEET_NAME); if (sh) delById(sh, data.id); } if (action === "saveSlot") { const sh = getOrCreate(ss, SLOTS_SHEET, ["ID","ชื่อ","วันที่","เริ่ม","สิ้นสุด","ประเภท","จำนวนสูงสุด","สถานที่"]); sh.appendRow([data.id,data.name,data.date,data.start, data.end,data.type,data.cap,data.location]); } if (action === "updateSlot") { const sh = ss.getSheetByName(SLOTS_SHEET); if (sh) { delById(sh, data.id); sh.appendRow([data.id,data.name,data.date,data.start, data.end,data.type,data.cap,data.location]); } } if (action === "deleteSlot") { const sh = ss.getSheetByName(SLOTS_SHEET); if (sh) delById(sh, data.id); const bsh = ss.getSheetByName(SHEET_NAME); if (bsh) delBySlot(bsh, data.id); } } function getOrCreate(ss,name,headers){ let sh=ss.getSheetByName(name); if(!sh){sh=ss.insertSheet(name);sh.appendRow(headers);} return sh; } function delById(sh,id){ const d=sh.getDataRange().getValues(); for(let i=d.length-1;i>=1;i--) if(String(d[i][0])===String(id)) sh.deleteRow(i+1); } function delBySlot(sh,sid){ const d=sh.getDataRange().getValues(); for(let i=d.length-1;i>=1;i--) if(String(d[i][5])===String(sid)) sh.deleteRow(i+1); } function fmtDate(v){ if(!v) return ''; if(v instanceof Date){ const y=v.getFullYear(); const m=String(v.getMonth()+1).padStart(2,'0'); const d=String(v.getDate()).padStart(2,'0'); return y+'-'+m+'-'+d; } return String(v); } function fmtTime(v){ if(!v) return ''; if(v instanceof Date){ const h=String(v.getHours()).padStart(2,'0'); const m=String(v.getMinutes()).padStart(2,'0'); return h+':'+m; } return String(v); } function out(obj, cb){ const json = JSON.stringify(obj); const body = cb ? cb+'('+json+')' : json; const mime = cb ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON; return ContentService.createTextOutput(body).setMimeType(mime); }
